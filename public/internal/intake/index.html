  <script>

    function v(id){ return (document.getElementById(id)?.value || "").trim(); }


    function buildText(){

      const stamp = new Date().toLocaleString();

      const source = v("source") || "—";

      const category = v("category") || "—";

      const priority = v("priority") || "—";

      const owner = v("owner") || "—";

      const summary = v("summary") || "—";


      return [

        "Square Bidness — Internal Intake",

        `Created: ${stamp}`,

        "",

        `Source: ${source}`,

        `Category: ${category}`,

        `Priority: ${priority}`,

        `Owner: ${owner}`,

        "",

        "Summary:",

        summary,

        "",

        "— /internal/intake/"

      ].join("\n");

    }


    function updateLinks(text){

      const subject = encodeURIComponent("Square Bidness | Internal Intake");

      const body = encodeURIComponent(text);


      const ops = "ops@squarebidness.com";

      const marcus = "marcusbuckner@squarebidness.com";


      document.getElementById("btnEmailOps").href = `mailto:${ops}?subject=${subject}&body=${body}`;

      document.getElementById("btnEmailMarcus").href = `mailto:${marcus}?subject=${subject}&body=${body}`;

    }


    function render(){

      const text = buildText();

      document.getElementById("preview").textContent = text;

      updateLinks(text);

    }


    ["source","category","summary","priority","owner"].forEach(id => {

      const el = document.getElementById(id);

      if (!el) return;

      el.addEventListener("input", render);

      el.addEventListener("change", render);

    });


    document.getElementById("btnCopy").addEventListener("click", async () => {

      const text = buildText();

      try{

        await navigator.clipboard.writeText(text);

        alert("Copied intake summary.");

      }catch(e){

        alert("Copy failed. You can select + copy from the preview box.");

      }

    });


    document.getElementById("btnClear").addEventListener("click", () => {

      if (!confirm("Clear this intake form?")) return;

      ["source","summary","owner"].forEach(id => document.getElementById(id).value = "");

      document.getElementById("category").value = "Web / Tech";

      document.getElementById("priority").value = "Normal";

      render();

    });


    // First paint

    render();

  </script>
